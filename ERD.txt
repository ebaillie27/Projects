Table users as U {
  user_id int [pk, increment] // auto-increment
  first_name varchar
  last_name varchar
}

Table customer as C {
  customer_id int [pk]
  user_id int [ref: > U.user_id]
  username text [Unique, not null]
  cus_password bytea
}

Table employee as E {
  employee_id int [pk]
  username text [Unique, not null]
  emp_password bytea
  user_id int [ref: > U.user_id]
  title_id int [ref: > T.title_id]
  location_id int [ref: > SL.location_id]
}

Table title as T {
  title_id int [pk]
  title varchar
}

Table store_location as SL {
  location_id int [pk]
  location_name varchar
}

Table supply {
  supply_id int [pk]
  item_id int [not null, ref: > I.item_id]
  quantity int
  location_id int [ref: > SL.location_id]
}

Table items as I {
  item_id int [pk]
  item_name varchar
  item_price money
}

Table purchase as Pu {
  purchase_id int [pk]
  customer_id int [ref: > C.customer_id]
  location_id int [ref: > SL.location_id]
  order_quantity int
  total_order currency
}
Table purchase_detail as OD {
  purchase_detail_id int [pk]
  purchase_id int [ref: > Pu.purchase_id]
  item_id int [ref: > I.item_id]
}
